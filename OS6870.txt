    function switchProject(id) {
        currentProject = id;
        document.getElementById('btn-os6900').className = id === 'os6900' ? 'nav-btn active' : 'nav-btn';
        document.getElementById('btn-os6870').className = id === 'os6870' ? 'nav-btn active' : 'nav-btn';

    function updateDashboard() {
        const ma = document.getElementById('ma-selector').value;
        const p = db[currentProject];
        const s = p[ma];

        document.getElementById('main-title').innerText = p.title;
        document.getElementById('label-old-year').innerText = p.oldYear;
        document.getElementById('val-old-price').innerText = '฿' + p.oldPriceNum.toLocaleString();
        document.getElementById('val-new-price').innerText = '฿' + p.newPriceNum.toLocaleString();
        document.getElementById('val-est-price').innerText = '฿' + s.est.toLocaleString();
        document.getElementById('val-diff-percent').innerText = s.diff;
        document.getElementById('val-diff-amount').innerText = '(' + s.amt + ')';

        const alertCard = document.getElementById('alert-card');
        alertCard.className = s.status === 'red' ? 'card alert-red' : 'card alert-green';
        
        const badge = document.getElementById('current-status-badge');
        badge.innerText = `ข้อมูล: ${p.title.split(' ')[1]} | สมมติฐาน MA ${ma}%`;
        badge.style.background = s.status === 'red' ? '#e53e3e' : '#38a169';

        document.getElementById('desc-old').innerHTML = s.descOld;
        document.getElementById('desc-est').innerHTML = s.descEst;
        document.getElementById('desc-new').innerHTML = s.descNew;

        myChart.data.datasets[0].data = s.dataOld;
        myChart.data.datasets[1].data = s.dataEst;
        myChart.data.datasets[2].data = s.dataNew;
        myChart.update();
    }

    function filterChart(idx) {
        if(idx === 'all') {
            myChart.setDatasetVisibility(0, true); myChart.setDatasetVisibility(1, true); myChart.setDatasetVisibility(2, true);
        } else {
            myChart.setDatasetVisibility(0, idx === 0);
            myChart.setDatasetVisibility(1, idx === 1);
            myChart.setDatasetVisibility(2, idx === 2);
        }
        myChart.update();
    }

    window.onload = updateDashboard;
